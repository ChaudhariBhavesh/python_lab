fruits=['Apple','mango','grapes']
fruits.append('Pears') #Append
print(fruits)

fruits.insert(0,'Banana') #Insert at 0th index
print(fruits)
fruits.insert(2,'Strawberry')
print(fruits)

more_fruits=['orange','kiwi']
fruits.extend(more_fruits) #Extend
print(fruits)

fruits.remove('Apple') #Remove
print(fruits)
